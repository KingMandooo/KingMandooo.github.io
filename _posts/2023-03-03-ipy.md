#include <stdio.h>

struct test

};

int main()

	ptr = &st;

    ptr->a = 5;        // 위와 동일
  
	(*ptr).b = 2;
	ptr->b = 3;        // 위와 동일

	printf("%d\n%d\n", (*ptr).a, (*ptr).b);

	return 0;
}
